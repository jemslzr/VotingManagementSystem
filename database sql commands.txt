CREATE DATABASE electiondb;

CREATE TABLE voters (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    address VARCHAR(255),
    date_of_birth DATE,
    last_vote_date DATE
);

CREATE TABLE candidates (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    position VARCHAR(100),
    party_list VARCHAR(100)
);

CREATE TABLE votes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    voter_name VARCHAR(100),
    candidate_name VARCHAR(100),
    voter_id INT,
    candidate_id INT,
    FOREIGN KEY (voter_id) REFERENCES voters(id),
    FOREIGN KEY (candidate_id) REFERENCES candidates(id)
);

CREATE TABLE archivedvoters (
    id INT,
    name VARCHAR(100),
    address VARCHAR(255),
    date_of_birth DATE,
    last_vote_date DATE,
    archive_date TIMESTAMP
);

CREATE TABLE archivedvotes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    voter_name VARCHAR(100),
    candidate_name VARCHAR(100),
    vote_date TIMESTAMP
);
